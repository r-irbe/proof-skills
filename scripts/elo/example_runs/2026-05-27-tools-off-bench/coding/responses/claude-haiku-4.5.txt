def count_inversions(arr: list[int]) -> int:
    def merge_sort(arr):
        if len(arr) <= 1:
            return arr, 0
        mid = len(arr) // 2
        left, left_inv = merge_sort(arr[:mid])
        right, right_inv = merge_sort(arr[mid:])
        merged = []
        inv = left_inv + right_inv
        i = j = 0
        while i < len(left) and j < len(right):
            if left[i] <= right[j]:
                merged.append(left[i])
                i += 1
            else:
                merged.append(right[j])
                inv += len(left) - i
                j += 1
        merged.extend(left[i:])
        merged.extend(right[j:])
        return merged, inv
    _, count = merge_sort(arr)
    return count

def is_balanced(s: str) -> bool:
    stack = []
    pairs = {'(': ')', '[': ']', '{': '}'}
    for char in s:
        if char in pairs:
            stack.append(char)
        elif char in pairs.values():
            if not stack or pairs[stack.pop()] != char:
                return False
    return len(stack) == 0

def longest_palindrome_subseq(s: str) -> int:
    if not s:
        return 0
    n = len(s)
    dp = [[0] * n for _ in range(n)]
    for i in range(n):
        dp[i][i] = 1
    for length in range(2, n + 1):
        for i in range(n - length + 1):
            j = i + length - 1
            if s[i] == s[j]:
                dp[i][j] = dp[i + 1][j - 1] + 2
            else:
                dp[i][j] = max(dp[i + 1][j], dp[i][j - 1])
    return dp[0][n - 1]

def fizzbuzz_count(n: int) -> int:
    if n <= 0:
        return 0
    count = 0
    for k in range(1, n + 1):
        div3 = k % 3 == 0
        div5 = k % 5 == 0
        if div3 != div5:
            count += 1
    return count

def min_jumps(arr: list[int]) -> int:
    if len(arr) <= 1:
        return 0
    if arr[0] == 0:
        return -1
    n = len(arr)
    jumps = 0
    current_max = 0
    next_max = 0
    for i in range(n - 1):
        next_max = max(next_max, i + arr[i])
        if i == current_max:
            if next_max <= current_max:
                return -1
            jumps += 1
            current_max = next_max
            if current_max >= n - 1:
                return jumps
    return jumps
